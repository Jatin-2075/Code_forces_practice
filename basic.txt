#used to count the smallest number prime
const int MAXA = 1e6 + 5;
int spf[MAXA];
void build_spf() {
    for (int i = 2; i < MAXA; i++) {
        if (spf[i] == 0) {
            for (long long j = i; j < MAXA; j += i)
                if (spf[j] == 0) spf[j] = i;
        }
    }
}
